Работа с базой данных
1 задание
SELECT c.login,
         COUNT(o."courierId")
FROM "Orders" as o
         LEFT OUTER JOIN "Couriers" as c
                    ON o."courierId" = c.id
WHERE o."inDelivery" = TRUE
GROUP BY c.login;

2 задание
SELECT DISTINCT track,
       CASE
           WHEN "finished"=TRUE THEN 2
           WHEN "cancelled"= TRUE then -1
           WHEN "inDelivery"=TRUE THEN 1
           ELSE 0
       END AS status
FROM "Orders";
